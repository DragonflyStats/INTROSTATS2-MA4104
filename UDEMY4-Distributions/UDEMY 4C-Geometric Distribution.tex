\frametitle{Geometric Distribution}
\Large
\[ P(X = n) = (1-p)^{n-1}p \]

\[ P(X > n) = (1-p)^n \]


\end{frame}
%--------------------------------------%
\begin{frame}
\frametitle{Geometric Distribution}
\Large
\begin{itemize}
\item The expeced value is $ E[X] = 1/p $

\item The variance of the geometric distribution is 
\[Var(X) = \frac{1-p}{p^2}\].

\end{itemize}
