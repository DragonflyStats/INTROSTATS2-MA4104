\begin{frame}
\frametitle{Chi-Square Test of Association}
\Large
\textbf{Cell$_{(1,1)}$}
\begin{itemize}
\item Row 1 : Row Total = 200
\item Column 1 : Column Total = 150
\item Overall Total = 600
\end{itemize}
\bigskip


Expected value for Cell$_{(1,1)}$

\[ E_{(1,1)} = \frac{200 \times 150}{600} = \frac{30,000}{600} = 50 \]
\end{frame}
